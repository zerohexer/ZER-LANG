# Fixed-buffer baseline — Stage 3 (2026-04-28)
#
# This file lists every fixed-size buffer declaration in compiler internals
# that has been REVIEWED and accepted. The audit script
# `tools/audit_fixed_buffers.sh` greps for fixed-size buffer patterns and
# diffs against this baseline. Anything new = lint failure.
#
# Format: file:line:source-line (verbatim from the .c file)
# Comments and blank lines are ignored.
#
# To add a new acceptable pattern, append the line + a justification comment
# above it explaining why the fixed size is safe (bounded by AST structure,
# short-lived scratch, etc.).
#
# To convert a pattern to dynamic, remove its line from here and use
# stack-first dynamic (see parser.c stack_iargs / zercheck_ir.c fast_stack).

# Bounded sets of "candidates to check" — count is fixed by AST structure
# (var-decl init OR fallback = 2; addr_exprs has 4 specific check kinds).
checker.c:            Node *arr_checks[2] = { NULL, NULL };
checker.c:                    Node *checks[2] = { node->var_decl.init, NULL };
checker.c:                    Node *addr_exprs[4] = { NULL, NULL, NULL, NULL };
checker.c:    Node *roots[2] = { value, NULL };
checker.c:                Node *roots[2] = { node->ret.expr, NULL };
checker.c:        Type *found[4];

# Stack-first dynamic — these have arena overflow (parser RF9 pattern).

parser.c:            Node *stack_vals[16];

# zerc_main.c import limit — 64 imported modules per compilation unit.
# Hard-coded by design: no real codebase imports >64 modules per file,
# and import depth is capped earlier by parser. Graceful degradation
# (silent skip beyond 64) confirmed acceptable in CLAUDE.md "Confirmed
# NOT Bugs" section.
zerc_main.c:    Node *import_asts[64];

# Capture-on-FIRE defer-body stack (plt86m defer-goto fix): stack-first per
# CLAUDE rule #7 — inline 32, arena-doubling overflow grown in the NODE_DEFER
# handler (never silently drops; defer_bodies_cap doubles). Mirrors LowerCtx's
# own label_inline[32] in the same struct.
ir_lower.c:    Node *defer_bodies_inline[32];
# B1 multi-root lock (2026-06-21): bounded local scratch for the distinct
# shared(rw) roots in ONE statement. Realistic max is 2-3 (a statement reading
# >16 distinct shared structs is not a real program); find_all_shared_roots_expr
# stops at max, never overflowing. emit_shared_lock_if_needed +
# emit_shared_unlock_if_needed re-derive the same set, so lock/unlock stay paired.
ir_lower.c:        Node *roots[16]; int n = 0;
